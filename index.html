const SHEET_ID = '19yDrHook-yOzy88Qs3m2QIAjoktfqkRF7cH4StpYSJA'; function doGet(e) { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0]; var data = sheet.getDataRange().getValues(); var json = JSON.stringify(data); var callback = e.parameter ? e.parameter.callback : null; if(callback) { return ContentService.createTextOutput(callback+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0]; var body = JSON.parse(e.postData.contents); var data = sheet.getDataRange().getValues(); if(body.action === 'add') { sheet.appendRow([body.id, body.fecha, body.tipo, body.pagador, body.categoria, body.subcategoria, body.detalle, body.importe, body.medio, body.compartido, body.quienDebe]); } else if(body.action === 'edit') { for(var i = 0; i < data.length; i++) { if(String(data[i][0]) === String(body.id)) { sheet.getRange(i+1, 1, 1, 11).setValues([[body.id, body.fecha, body.tipo, body.pagador, body.categoria, body.subcategoria, body.detalle, body.importe, body.medio, body.compartido, body.quienDebe]]); break; } } } else if(body.action === 'delete') { for(var j = 0; j < data.length; j++) { if(String(data[j][0]) === String(body.id)) { sheet.deleteRow(j+1); break; } } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); }
